async def process_video_generation(job_id: str, req: VideoRequest, base_url: str):
    try:
        # 🟢 STEP 1: Update Status
        video_jobs[job_id] = {"status": "processing", "progress_msg": "📝 Writing Lesson Script..."}
        
        strict_prompt = f"Teacher for {req.class_name}, Chapter: {req.chapter_name}, Focus: {req.specific_topic}"
        script = await generate_script_with_gemini(strict_prompt, req.duration_minutes)
        paragraphs = [p.strip() for p in script.split('\n\n') if p.strip()]

        # 🟢 STEP 2: Update Status
        video_jobs[job_id]["progress_msg"] = f"🎙️ Generating Voiceover (0/{len(paragraphs)} slides)..."
        
        base_project_dir = Path(__file__).resolve().parent.parent.parent
        temp_dir = str(base_project_dir / "static_school" / "temp_videos")
        os.makedirs(temp_dir, exist_ok=True)
        
        chunks_data = []
        for index, para in enumerate(paragraphs):
            # Update percentage per slide
            video_jobs[job_id]["progress_msg"] = f"🎨 Creating Slides & Audio ({index + 1}/{len(paragraphs)})..."
            
            audio_path = os.path.join(temp_dir, f"audio_{job_id}_{index}.mp3")
            img_path = os.path.join(temp_dir, f"img_{job_id}_{index}.png")

            await create_voiceover(para, audio_path)
            create_slide_image(para, img_path, index + 1, len(paragraphs))
            chunks_data.append((img_path, audio_path))

        # 🟢 STEP 3: Update Status
        video_jobs[job_id]["progress_msg"] = "🎬 Stitching Final Video (FFMPEG)..."
        
        temp_filename = f"teacher_{req.teacher_id}_{job_id}_preview.mp4"
        final_video_path = os.path.join(temp_dir, temp_filename)
        await asyncio.to_thread(stitch_dynamic_video, chunks_data, final_video_path)

        # 🟢 STEP 4: Final Success
        video_jobs[job_id] = {
            "status": "completed",
            "progress_msg": "✅ Ready!",
            "preview_url": f"{base_url}/static_school/temp_videos/{temp_filename}",
            "temp_file_name": temp_filename
        }

    except Exception as e:
        video_jobs[job_id] = {"status": "failed", "progress_msg": "❌ Error occurred", "error": str(e)}


# ✅ ADDED: duration_minutes parameter to the function
async def generate_script_with_gemini(topic_prompt: str, duration_minutes: int) -> str:
    """Asks Gemini to write an educational script based on the teacher's prompt and time limit."""
    
    # ⏱️ Calculate words (Average speaking rate is ~130 words per minute)
    target_word_count = duration_minutes * 130
    example_count = max(1, duration_minutes - 1) # 1 min = 1 example, 5 mins = 4 examples

    model = genai.GenerativeModel('gemini-flash-latest')
    instruction = f"""
    You are an expert, friendly CBSE Math teacher at St. Anthony's School.
    Write a spoken script for a video lesson based on this exact topic request: "{topic_prompt}"
    
    Rules:
    1. The script must be roughly {target_word_count} words long (to make a {duration_minutes}-minute video).
    2. Speak directly to the students in a warm, encouraging tone.
    3. CRITICAL: Break the script into SHORT paragraphs. Maximum 2 to 3 sentences per paragraph. (Each paragraph becomes one presentation slide, so they cannot be too long).
    4. Provide at least {example_count} real-world example(s) in the script. 
    5. The example MUST start exactly with the word "Example: " on a new line.
    6. Do NOT use markdown formatting like **bold** or *italics*.
    7. Do NOT use emojis.
    """
    try:
        response = model.generate_content(instruction)
        print(f"✅ Gemini Response generated successfully.")
        print(f"Gemini Response {response.text}")
        return response.text
    except ResourceExhausted:
        print("❌ Gemini API Rate Limit Hit!")
        # This will send a clean 429 error back to Flutter instead of crashing!
        raise HTTPException(
            status_code=429, 
            detail="The AI is currently generating too many lessons. Please wait 1 minute and try again."
        )

async def create_voiceover(text: str, output_audio_path: str):
    """Generates audio for a specific paragraph"""
    communicate = edge_tts.Communicate(text, "en-US-JennyNeural")
    await communicate.save(output_audio_path)



def create_slide_image(paragraph_text: str, output_image_path: str, slide_number: int, total_slides: int):
    """Generates a Blackboard image with Large Fonts and Yellow Examples"""
    # Purple Theme: 1280 width / 2 = 640 (Center X)
    img = Image.new('RGB', (1280, 720), color=(94, 23, 131)) 
    draw = ImageDraw.Draw(img)
    
    try:
        # ✅ Automatically pick the right font based on the operating system
        if platform.system() == "Windows":
            # Running on your local D:\ drive
            font = ImageFont.truetype("arialbd.ttf", 60)  # Arial Bold, Size 150
            small_font = ImageFont.truetype("arial.ttf", 40)
        else:
            # Running on your AWS Ubuntu server
            font = ImageFont.truetype("/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf", 60)
            small_font = ImageFont.truetype("/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf", 40)
    except Exception as e:
        print(f"FONT ERROR: {e}")
        font = ImageFont.load_default()
        small_font = font

    if "Example:" in paragraph_text:
        parts = paragraph_text.split("Example:")
        main_text = parts[0].strip()
        example_text = "Example: " + parts[1].strip()
    else:
        main_text = paragraph_text.strip()
        example_text = ""

    # ✅ Reduced wrap width to 16 because the letters are now massive!
    wrapped_main = "\n".join(textwrap.wrap(main_text, width=40))
    
    draw.multiline_text(
        (640, 60),                  # Raised starting Y to 60 to fit the huge text
        wrapped_main, 
        fill=(255, 255, 255), 
        font=font, 
        spacing=20,
        align="center", 
        anchor="ma" 
    )
    
    if example_text:
        wrapped_example = "\n".join(textwrap.wrap(example_text, width=40))
        
        # ✅ DYNAMIC SPACING: Calculate exactly where the main text ended
        bbox = draw.multiline_textbbox((640, 60), wrapped_main, font=font, spacing=20, align="center", anchor="ma")
        y_bottom_of_main = bbox[3] 
        
        # Start the example 40 pixels below the main text
        y_start_for_example = y_bottom_of_main + 40 
        
        draw.multiline_text(
            (640, y_start_for_example), 
            wrapped_example, 
            fill=(255, 235, 59), 
            font=font, 
            spacing=20,
            align="center", 
            anchor="ma" 
        )
    
    draw.text((1050, 650), f"Slide {slide_number} of {total_slides}", fill=(200, 200, 200), font=small_font)
    img.save(output_image_path)


def stitch_dynamic_video(chunks_data: list, output_video_path: str):
    clips = []
    
    for img_path, audio_path in chunks_data:
        audio = AudioFileClip(audio_path)
        img = ImageClip(img_path).set_duration(audio.duration)
        video_chunk = img.set_audio(audio)
        clips.append(video_chunk)
    
    final_video = concatenate_videoclips(clips, method="compose")
    base_dir = Path(__file__).resolve().parent.parent.parent
    robot_gif_path = os.path.join(base_dir, "robot.gif")
    
    if os.path.exists(robot_gif_path):
        print(f"🤖 Found avatar at {robot_gif_path}! Overlaying...")
        robot_clip = VideoFileClip(robot_gif_path, has_mask=True)
        robot_clip = robot_clip.fx(vfx.loop, duration=final_video.duration)
        robot_clip = robot_clip.resize(height=280).set_position(("right", "bottom"))
        final_video = CompositeVideoClip([final_video, robot_clip])
    
    final_video.write_videofile(
        output_video_path, 
        fps=24, 
        codec="libx264", 
        audio_codec="aac",
        ffmpeg_params=["-pix_fmt", "yuv420p"] 
    )
    
    for c in clips:
        c.close()
    if 'robot_clip' in locals():
        robot_clip.close()
